#include <stdio.h>
#include <stdlib.h>
int conta_occorrenze(int *arr, int pos, int dimArr) {
	int i, c;
	c=0;

	for(i=0; i<dimArr; i++) {
		if(arr[i] == arr[pos]) {
			c++;
		}
	}
	return c;
}
